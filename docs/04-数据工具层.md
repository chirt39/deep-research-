def _duckduckgo_search_records(query: str, count: int = 8) -> list[dict]:
    """DuckDuckGo 免费网页搜索。

    1. 使用 ddgs 库（pip install ddgs）
    2. DDGS().text(query, max_results=count) 返回生成器
    3. 每条结果提取：title, href→url, body→snippet
    4. 自动提取域名：_extract_domain(href)
    5. 分配 source_id：DDG-{序号}
    
    错误处理：
    - ImportError（ddgs 未安装）→ 警告 + 返回 []
    - 网络错误 → 警告 + 返回 []
    - 永不抛异常，保证工作流不中断
    
    ⚠️ DuckDuckGo 在中国大陆可能不稳定，
    建议配 BOCHA_API_KEY 走 Bocha 搜索。
    """

def _extract_domain(url: str) -> str:
    """从 URL 提取域名。
    "https://cj.sina.com.cn/article/123" → "cj.sina.com.cn"
    """

def bocha_web_search_records(query: str, count: int = 8) -> list[dict]:
    """网页搜索主入口。
    
    策略：
    ① os.getenv("BOCHA_API_KEY") → 有配？
    ② 是 → 调用 Bocha API（POST https://api.bocha.cn/v1/web-search）
       - payload: {query, summary:True, freshness:"noLimit", count}
       - headers: Authorization: Bearer {api_key}
       - 响应解析：data.webPages.value[] 或其他嵌套格式
       - 每条提取：name→title, url, summary→snippet, 域名提取
       - 分配 source_id：WEB-{序号}
    ③ 否 → 降级到 _duckduckgo_search_records(query, count)
    """

## 4.3 AKShare 金融数据

### 4.3.1 基础设施

```python
_AKSHARE_AVAILABLE = False
try:
    import akshare as ak
    _AKSHARE_AVAILABLE = True
except ImportError:
    # 未安装时不报错，所有函数返回 [{"error": "AKShare 未安装"}]

def _safe_finance_call(func, *args, **kwargs):
    """安全调用 AKShare 函数。
    ① _AKSHARE_AVAILABLE → False → 返回 (None, "AKShare 未安装")
    ② 调用 func(*args, **kwargs)
    ③ 异常 → 返回 (None, "金融数据获取失败: {exc}")
    ④ 成功 → 返回 (result, None)
    
    设计意图：AKShare 的 API 不稳定（版本更新频繁），
    用这个包装让所有调用统一降级。
    """

def _normalize_stock_code(code: str) -> str:
    """标准化股票代码。
    "002594" → "sz002594"    （0/3 开头 → 深交所）
    "600519" → "sh600519"    （6/9 开头 → 上交所）
    已有前缀的保持不变
    """

def _detect_stock_code(query: str) -> str | None:
    """从用户查询中提取 6 位股票代码。
    正则：r'(?<!\d)(\d{6})(?!\d)'
    "比亚迪002594怎么样" → "sz002594"
    "分析茅台600519" → "sh600519"
    未找到 → None
    """

def _parse_pct(value) -> float:
    """解析百分比/大数字符串。
    "55.38%" → 55.38
    "-11.82%" → -11.82
    "40.85亿" → 4085000000.0
    "1502.25万" → 15022500.0
    纯数字直接转 float
    """
```

### 4.3.2 数据采集函数

def fetch_stock_price_records(symbol, period="monthly", count=24) -> list[dict]:
    """获取历史行情。
    
    调用：ak.stock_zh_a_hist(symbol=code, period=period, adjust="qfq")
    
    输出每条记录：
    {source_id, title, snippet, date, url:"",
     source_type:"finance_data", data_type:"price_history"}
    
    关键字段：date/opening/close/high/low/volume
    """

def fetch_stock_info_record(symbol) -> dict | None:
    """获取公司基本信息。
    
    调用：ak.stock_individual_info_em(symbol=symbol)
    
    输出：
    {name, pe, pb, total_mv, industry, listing_date, total_shares}
    
    ⚠️ 该 API 连接 EastMoney push2 端点，部分网络环境可能被拦
    （和 stock_zh_a_spot_em 一样的问题）
    """

def fetch_financial_indicators_records(stock_code) -> list[dict]:
    """获取财务指标（同花顺数据源）。
    
    调用：ak.stock_financial_abstract_ths(symbol=short_code, indicator="按报告期")
    
    使用同花顺而非东方财富的原因：
    - EastMoney push2 端点在某些网络环境被拦
    - 同花顺数据端点在同样的网络环境下能通
    
    提取指标（共 20+ 个字段，选最核心的 8 个）：
    - 净资产收益率 (ROE)
    - 销售毛利率 / 销售净利率
    - 营业总收入同比增长率 / 净利润同比增长率
    - 资产负债率
    - 基本每股收益 (EPS) / 每股净资产
    
    输出每条记录：
    {source_id, title, snippet, url:"", report_date,
     source_type:"finance_data", data_type:"financial_indicators"}
    """

def fetch_stock_news_records(stock_code, limit=10) -> list[dict]:
    """获取个股新闻（东方财富）。
    
    调用：ak.stock_news_em(symbol=code)

    {source_id:"FIN-NEWS-{n}", title, url, snippet, publish_time,

## 4.4 数据流向（以"比亚迪002594"为例）

```
web_search_node 执行：

① bocha_web_search_records("比亚迪002594")
   → DuckDuckGo 搜索 → 5 条 DDG 记录
   
② collect_finance_data_records("比亚迪002594")
   ├─ fetch_stock_info_record("sz002594")
   │  → 成功：{"name":"比亚迪","pe":25.0,"pb":4.5,...}
   │     └─→ FIN-INFO-1
   │  → 失败：None（被代理拦截时）
   │
   ├─ fetch_financial_indicators_records("002594")
   │  → stock_financial_abstract_ths("002594", "按报告期")
   │  → 8 条季度指标（最近 2 年）
   │     └─→ FIN-IND-1 ~ FIN-IND-8
   │
   └─ fetch_stock_news_records("002594", limit=8)
      → stock_news_em(symbol="002594")
      → 8 条新闻
         └─→ FIN-NEWS-1 ~ FIN-NEWS-8

③ 合并：5 + 0~1 + 8 + 8 = 21~22 条 raw_records

④ LLM 过滤 → web_evidence[] → deep_dive → source_index[] → 参考资料
```
