## 7.1 入口文件（app_main.py）

```python
# 文件: app/app_main.py

def create_app() -> FastAPI:
    """创建 FastAPI 应用实例。
    
    流程：
    ① AppSettings() — 加载应用配置（端口、CORS 等）
    ② FastAPI(title="DeepResearch Multi-Agent Assistant")
    ③ 添加 CORS 中间件
       - allow_origins = ["http://localhost:5173", "http://127.0.0.1:5173"]
         ↑ Vite 开发服务器地址
       - allow_credentials=True（支持 Cookie）
       - allow_methods=["*"]（所有 HTTP 方法）
       - allow_headers=["*"]（所有请求头）
    ④ 注册路由
       - health_router（/health）
       - research_router（/api/v1/research/*）
    """

app = create_app()  # 模块级单例（uvicorn 导入时即创建）

if __name__ == "__main__":
    # 直接运行 python app_main.py 时启动
    uvicorn.run("app_main:app", host="0.0.0.0", port=8000,
                reload=settings.app_env=="development")
```

```python
# 文件: app/backend/router/research_router.py

router = APIRouter(prefix="/api/v1/research", tags=["research"])

@router.post("/run", response_model=ResearchResponse)
async def run_research(payload: ResearchRequest,
                        workflow_service=Depends(get_workflow_service)):
    """POST /api/v1/research/run
    
    同步接口：等待研究完成后一次性返回结果。
    
    请求体：
    {
      "query": "深度研究比亚迪002594",
      "user_id": "user01",          // 默认 default_user
      "thread_id": "thread01",      // 默认 default_thread
      "tenant_id": "default_tenant",
      "max_iterations": null,       // null = 使用 config.json 默认值
      "enable_memory": null         // null = 使用 config.json 默认值
    }
    
    响应：
    {
      "query": "...",
      "user_id": "user01",
      "thread_id": "thread01",
      "tenant_id": "default_tenant",
      "final": "## 投资摘要\n..."   // 完整 Markdown 研报
    }
    """

@router.post("/stream")
async def stream_research(payload: ResearchRequest,
                           workflow_service=Depends(get_workflow_service)):
    """POST /api/v1/research/stream
    
    流式接口：通过 Server-Sent Events 实时推送执行进度。
    
    响应格式（SSE）：
    Content-Type: text/event-stream
    
    data: {"type":"status","message":"任务已接收..."}
    data: {"type":"phase","node":"intent","message":"Intent Router 正在识别"}
    data: {"type":"phase","node":"plan","message":"Planner 正在拆解"}
    data: {"type":"phase","node":"web_search","message":"Web Scout 正在检索"}
    ...
    data: {"type":"route","message":"已走多智能体研究路径"}
    data: {"type":"final","query":"...","final":"## 投资摘要\n..."}
    
    事件类型：
    - status:  任务状态
    - phase:   节点执行进度（node + message）
    - route:   路由选择（direct/multiagent）
    - final:   最终结果（含完整研报）
    - error:   错误信息
    """
```

## 7.5 WorkflowService（核心桥接层）

```python
# 文件: app/backend/service/workflow_service.py

class WorkflowService:
    """工作流服务 —— 连接 FastAPI 和多智能体引擎的桥梁。
    
    采用懒初始化 + 线程安全 + 单例模式。
    """

    def __init__(self, config_path: str):
        self._config_path = config_path     # config.json 路径
        self._lock = Lock()                 # 线程安全锁
        self._initialized = False           # 懒初始化标记
        self._base_config = None            # 基础配置
        self._memory_manager = None         # 记忆管理器
        self._app = None                    # LangGraph 应用

    def _ensure_initialized(self):
        """懒初始化（首次调用时触发）。
        
        使用双重检查锁定模式：
        if not _initialized:
            with lock:
                if not _initialized:
                    ① AppConfig.from_file(config_path)
                    ② build_memory_manager(config)
                    ③ build_agents(config.model, config.api_key, config.base_url, config)
                    ④ build_checkpointer(config)
                    ⑤ build_workflow_app(agents, checkpointer)
                    → _initialized = True
        
        只初始化一次，后续所有请求复用同一个 app 实例。
        """

    def _build_runtime_config(self, user_id, thread_id, tenant_id,
                               max_iterations, enable_memory) -> AppConfig:
        """根据请求参数构建运行时配置。
        
        基础配置来自 config.json，user_id/thread_id 等用请求参数覆盖。
        使用 with_overrides() 创建新的不可变实例。
        """

    def _run_sync(self, query, user_id, thread_id, tenant_id,
                   max_iterations, enable_memory) -> tuple[str, str]:
        """同步执行研究（在线程池中运行）。
        
        流程：
        ① 确保已初始化
        ② 构建运行时配置
        ③ 注入记忆上下文（如有）
        ④ create_initial_state()
        ⑤ self._app.invoke(state, config)  ← LangGraph 同步执行
        ⑥ persist_turn() 持久化本轮对话
        ⑦ 返回 (final, route)
        """

    def _run_sync_with_events(self, query, ..., emit) -> tuple[str, str]:
        """同步执行 + 事件推送（用于 SSE 流式）。
        
        与 _run_sync 的区别：
        - 使用 self._app.stream(state, config, stream_mode="updates")
          而不是 self._app.invoke()
        - 每完成一个节点就调用 emit({"type":"phase","node":...,"message":...})
        - 前端可以在每个节点完成时看到进度更新
        
        stream_mode="updates" 的含义：
        - LangGraph 每执行完一个节点就 yield 一次
        - 每次 yield 的是 {node_name: node_output_dict}
        """

    # ── 异步接口（给 FastAPI 路由调用） ──

    async def run(self, query, ...):
        """异步包装：在线程池中执行同步研究，不阻塞事件循环。"""

    async def run_with_route(self, query, ...):
        """同 run()，额外返回路由选择。"""

    async def stream_events(self, query, ...) -> AsyncIterator[dict]:
        """SSE 事件流生成器。
        
        通过 asyncio.Queue 桥接同步线程和异步事件循环：
        - 同步线程 → queue.put(event)
        - 异步协程 → await queue.get() → yield event
        
        使用 Thread(target=worker, daemon=True) 启动同步工作线程。
        """
```

### 单例工厂

```python
# 文件: app/backend/service/__init__.py

@lru_cache(maxsize=1)
def get_workflow_service() -> WorkflowService:
    """单例工厂 —— 整个进程生命周期内只创建一次 WorkflowService。
    
    lru_cache(maxsize=1) 保证了：
    - 第一次调用 → 创建实例
    - 后续调用 → 返回同一实例
    - FastAPI 的 Depends() 每次请求都会调用此函数
    """
    settings = AppSettings()
    return WorkflowService(config_path=settings.config_path)
```

```
浏览器 "深度研究比亚迪002594"
  │
  │ POST /api/v1/research/stream
  │ Body: {query, user_id, thread_id, tenant_id}
  │
  ▼
research_router.stream_research()
  │
  │ Depends(get_workflow_service) → 获取单例 WorkflowService
  │
  ▼
WorkflowService.stream_events()
  │
  │ 创建 asyncio.Queue + Thread
  │
  ├─ 同步线程 ──────────────────────┐
  │  _ensure_initialized()          │
  │  _build_runtime_config()        │
  │  build_personalized_prompt_context() ← 记忆注入
  │  create_initial_state()         │
  │  app.stream(state, config) ─────┤ LangGraph 逐步执行
  │  │   node: intent               │
  │  │   emit({"type":"phase",...})  │
  │  │   node: plan                 │
  │  │   emit(...)                  │
  │  │   ...                        │
  │  │   node: write                │
  │  │   emit({"type":"final",...})  │
  │  persist_turn()                 │
  │  emit({"type":"__done__"})      │
  └─────────────────────────────────┘
  │
  │ 异步协程
  │ await queue.get() → yield SSE event
  │ ...
  │
  ▼
浏览器接收 SSE 事件流
  │
  ├─ {type:"phase", message:"Planner 正在拆解问题"}
  ├─ {type:"phase", message:"Web Scout 正在检索网络证据"}
  ├─ ...
  └─ {type:"final", final:"## 投资摘要\n..."}
     → 渲染 Markdown → 展示给用户
```
