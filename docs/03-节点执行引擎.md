```python
def _invoke_json_agent(state, prompt, agent, agent_name, node, fallback) -> tuple[dict, str, list]:
    """统一的 JSON 模式 LLM 调用。
    
    这是所有节点与 LLM 交互的标准入口，封装了：
    1. 记忆上下文注入：with_memory_context(state, prompt)
    2. Agent 调用：agent.invoke({"messages": [human]})
                    ↑ 注意：不传 state["messages"]，避免 token 累积
    3. JSON 解析：_extract_json_block + json.loads
    4. 日志输出：工具调用摘要 + LLM 输出预览
    
    返回：(解析后的 dict, 原始文本, messages 列表)
    
    重要设计：agent.invoke 只传 [human] 不传历史消息
    → 每个节点独立，不累积上下文
    → 节省 token（历史消息可能很长）
    → 依赖 state 传结构化数据，不依赖对话历史
    """

def detect_intent(query: str) -> str:

    两阶段匹配：
    
    阶段 1 — 强信号（直接判 multiagent）：
    - "20XX年" + 趋势/新闻/调研 → multiagent
    - 强关键词命中 → multiagent
    
    阶段 2 — 普通信号：
    - 22 个关键词：调研/研究/分析/方案/对比/报告/代码/数据/模型...

    - 全部未命中 → direct
    
    设计意图：金融类查询几乎 100% 命中 multiagent，

    """

def intent_node(state, agent, agent_name) -> ResearchState:

    1. detect_intent() 规则初判

    3. 只有判为 direct 时才调 LLM 二次确认（避免误判简单问题）
    
    返回：{"intent": "direct|multiagent", "draft": "", "messages": []}

def _default_plan(state) -> dict:

    生成一个包含原始问题的最小化大纲，
    确保即使 LLM 返回格式错误，工作流也能继续。
    """

def _derive_direct_search_queries(query) -> list[str]:
    """从原始查询自动生成搜索词（绕过 LLM）。
    
    策略：
    1. 提取核心实体（英文术语或中文实体名）
    2. 生成 6 种变体："{实体}是什么" "{实体} GitHub" "{实体} 官方文档" 等
    3. 去重，最多 6 条
    
    作用：即使 LLM 规划失败，系统也能生成基本的搜索词继续执行。
    """

def _derive_search_plan(outline, sub_questions, research_questions, query) -> list[dict]:
    """从 LLM 生成的 outline 中提取可执行的搜索计划。
    
    流程：
    1. 先加入直接推导的搜索词（_derive_direct_search_queries）
    2. 遍历 outline 中每个 section 的 search_queries
    3. 用地基检查过滤：必须与原始查询有术语重叠
    4. 去重，最多 6 条
    """

def plan_node(state, agent, agent_name) -> ResearchState:
    """规划节点 —— 将用户问题拆解为结构化研究计划。
    
    LLM 输入：用户原始 query
    LLM 输出：{objective, sub_questions[5个], outline[7节], search_plan, budget}
    
    后处理：
    1. 提取 outline → 生成具体搜索计划

## 3.5 网络搜索节点

```python
def _build_queries(state, source_preference) -> list[dict]:
    """构建当前节点需要的搜索查询。
    
    如果是补搜轮次（iteration > 0），用 supplementary_queries
    否则用 search_plan 中的查询。
    按 source_preference（web/local/hybrid）过滤，最多 6 条。
    """

def web_search_node(state, agent, agent_name) -> ResearchState:
    """网络搜索节点 —— 最复杂的节点。
    
    完整执行流程：
    
    ① 构建查询列表
       _build_queries(state, "web") → 最多 6 条查询
    
    ② 执行 Bocha/DuckDuckGo 网页搜索
       for each query:
           bocha_web_search_records(query, count=4)
           → BOCHA_API_KEY 有配? → Bocha API
           → 未配? → DuckDuckGo 免费搜索
           → 每条结果分配 source_id (WEB{迭代}_{序号}-{索引})
    
    ③ 执行 AKShare 金融数据采集
       collect_finance_data_records(state["query"])
       → 检测股票代码 002594
       → 新闻(stock_news_em) + 财务(stock_financial_abstract_ths)
       → 每条结果分配 source_id (FIN-{类型}-{序号})
    
    ④ 数据清洗
       _dedupe_sources(raw_records, ["url", "title"])    ← URL+标题去重
       _minimal_record_filter(raw_records, ["title","snippet","url"]) ← 过滤空记录
    
    ⑤ LLM 证据过滤
       将清洗后的记录发给 LLM（WebScout Agent）
       → LLM 判断相关性 + 信源分级
       → 返回 {evidence: [...], rejected_source_ids: [...]}
       
       防护机制：
       - LLM 返回空 evidence → 自动启用 fallback（所有记录视为证据）
       - _prune_evidence_to_allowed_sources() 确保 LLM 不编造 source_id
       - _enrich_evidence_from_raw() 从原始记录补充丢失的 url/domain
    
    ⑥ 输出
       - web_evidence[]: 过滤后的证据列表
       - web_search_trace[]: 每次查询的详细审计记录
       - web_retrieval_stats: 检索统计
    """

## 3.7 证据裁判节点

```python
def _score_evidence(record) -> tuple[float, str]:
    """对单条证据打分。
    
    评分规则：
    - source_type == "local"        → 0.92（企业内部知识库，默认高可信）
    - domain 是 .gov.cn / .gov / .edu → 0.88（官方/权威机构）
    - domain 含 news/finance/reuters 等 → 0.72（主流媒体）
    - domain 非空普通域名             → 0.58（需交叉验证）
    - domain 为空                     → 0.45（来源信息不完整）
    """

def _fallback_audit(state) -> dict:
    """证据裁判失败时的兜底方案。
    
    遍历 web_evidence + local_evidence：
    1. _score_evidence() 自动评分
    2. score >= 0.6 → 加入 source_index（可信来源）
    3. score < 0.6  → 加入 audit_flags（标记低置信）
    4. 构建完整的 evidence_pool
    """

def deep_dive_node(state, agent, agent_name) -> ResearchState:
    """证据裁判节点 —— 评分、去重、冲突审计。
    
    输入：web_evidence[] + local_evidence[]
    （两个并行检索节点在 graph 中汇聚到这个节点）
    
    LLM 处理：
    1. 评分：按 5 级标准打分（0.95-1.00 官方 → <0.40 匿名）
    2. 去重：相同 URL/内容合并
    3. 冲突审计：
       - 财务数据差异 >5% → 标记 conflict
       - 来源可信度低 → 标记 low_confidence
       - 信息缺失 → 标记 missing_evidence
       - 数据过时 → 标记 outdated
       - 利益冲突（券商是承销商）→ 标记 potential_bias
    
    后处理：
    - 截断 evidence_pool，LLM 漏掉的记录用 _score_evidence 补充
    - 生成 source_index（给 write_node 做引用校验）
    
    输出：evidence_pool[], source_index[], audit_flags[]
    """
```

def _fallback_analysis(state) -> dict:

def analyze_node(state, agent, agent_name) -> ResearchState:
    """分析节点 —— 从证据池形成投资结论。
    
    LLM 执行五维分析：
    1. 宏观面：货币政策/经济周期/外部环境
    2. 行业面：赛道景气度/竞争格局/政策催化
    3. 公司面：商业模式/财务健康度/管理层
    4. 估值面：PE/PB/PS/PEG vs 同业 vs 历史分位
    5. 舆情面：机构评级/资金流向/事件驱动
    
    每个维度的 confidence 基于证据充分度：
    - high：≥3 独立高质量来源
    - medium：2 个来源或 1 个高质量
    - low：仅 1 个来源
    
    输出：
    - findings[]: 每条结论 + 维度 + 置信度 + source_ids
    - needs_more_research: 是否需要补搜
    - missing_gaps: 缺失的信息维度列表
    """

## 3.10 报告撰写节点

```python
def _extract_citation_ids(content) -> list[str]:
    """从正文中提取所有引用 ID。
    
    正则：\[([A-Z]+\d+_\d+-\d+)\]
    匹配格式：WEB1_1-1, LOC1_1-3, FIN-IND-1, DDG-1 等
    
    返回去重后的 ID 列表（保持出现顺序）。
    """

def _validate_and_fix_citations(content, valid_source_ids) -> tuple[str, list[str]]:
    """校验并清理正文中的引用标记。
    
    问题：LLM 有时会编造不存在的 source_id
    解决：遍历所有 [XXX] 标记，不在 valid_source_ids 中的直接移除
    
    返回：(修正后的内容, 实际使用的合法引用列表)
    """

def _build_source_lookup(state) -> dict[str, dict]:
    """构建来源查找表。
    
    从四个来源汇总：
    - source_index[]（deep_dive 产出）
    - evidence_pool[]（deep_dive 产出）
    - web_evidence[]（web_search 产出）
    - local_evidence[]（local_rag 产出）
    
    合并规则：
    - 同一 source_id → locator/label/source_type 后到的补充而不是覆盖
    - LOC 前缀 → source_type 强制为 "local"
    - WEB 前缀 → source_type 强制为 "web"
    
    返回：{source_id: {source_id, source_type, label, locator}}
    """

def _render_reference_list(state) -> str:
    """渲染参考资料列表。
    
    引用收集优先级：
    1. 从 draft 正文中按出现顺序提取引用 ID
    2. 如果正文无引用 → 降级到 findings 中的 source_ids
    3. 如果仍为空 → 降级到全部 lookup keys
    
    展示规则：
    - web 来源在前，local 在后
    - 同一文件路径的 local 来源只展示一次（去重）
    - locator 为空 → "链接暂不可用" 或 "本地知识库"
    
    输出格式：
    ## 参考资料
    - [WEB1_1-1] [web]: 标题 | https://...
    - [DDG-1] [web]: 标题 | https://...
    - [FIN-IND-1] [finance_data]: 财务指标 2026Q1 ROE=1.7% 毛利率=18.8% |
    - [LOC1_1-1] [local]: 比亚迪2025年报.txt | /docs/比亚迪2025年报.txt
    """

def _ensure_reference_section(content, state) -> str:
    """确保报告末尾有参考资料章节。
    
    如果 LLM 生成的正文已包含 "## 参考资料" 或 "## 引用列表" → 不重复添加
    否则自动拼接 _render_reference_list()
    """

def write_node(state, agent, agent_name) -> ResearchState:
    """报告撰写节点 —— 生成最终研报。
    
    流程（7 步）：
    
    ① 构建合法引用 ID 列表
       valid_source_ids = source_index[] 中所有 source_id（最多 80 个）
    
    ② 组装 Prompt
       query + sub_questions + findings[] + source_index[] +
       valid_source_ids + audit_flags[]
       同时明确指令："禁止编造引用序号，只使用合法 ID"
    
    ③ 调用 LLM（Writer Agent）
       agent.invoke({"messages": [human]})  ← 不传历史消息
       生成 3000+ 字 Markdown 研报
    
    ④ 清理输出格式
       去掉 ```json、```markdown、``` 等包裹
    
    ⑤ 校验引用（防 LLM 编造）
       _validate_and_fix_citations() → 移除不在 valid_source_ids 中的引用
    
    ⑥ 拼接参考资料
       _ensure_reference_section() → 追加 ## 参考资料
    
    ⑦ 输出
       state["draft"] = state["final"] = 最终研报字符串
    """
```
