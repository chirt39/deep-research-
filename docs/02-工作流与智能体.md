## 2.1 State（共享状态）

```python
# 文件: app/mult_agents/state.py

class ResearchState(TypedDict):
    """多智能体共享状态。LangGraph 每个节点读写这个 dict。
    
    设计原则：
    - 每个字段有单一生产者（某个节点负责写入）
    - 多个消费者可以读取
    - messages 字段用 operator.add 自动累加（Annotated）
    """

    # ── 输入 ──
    query: str              # 用户原始问题

    # ── 多租户/记忆 ──
    user_id: str            # 用户标识
    tenant_id: str          # 租户标识
    memory_context: str     # 从记忆系统注入的上下文文本

    # ── 消息累积（特殊：Annotated[List, operator.add]） ──
    messages: Annotated[List[BaseMessage], operator.add]
    # 每次节点返回 {"messages": [...]} 时自动追加，不覆盖

    # ── 意图 ──
    intent: str             # "direct" | "multiagent"
    phase: str              # 当前阶段描述

    # ── 规划 (plan_node) ──
    plan: str               # 规划摘要文本
    outline: list[dict]     # 大纲结构 [{id, title, search_queries, ...}]
    sub_questions: list[str]# 拆解出的子问题
    research_questions: list[str]  # 研究问题列表
    search_plan: list[dict] # 搜索计划 [{section_id, query, source_preference}]
    budget: dict            # 预算限制 {max_rounds, max_sources, max_tokens, max_seconds}

    # ── 检索结果 (web_search_node / local_rag_node) ──
    web_search: str         # 网络检索摘要
    local_rag: str          # 本地知识库检索摘要
    web_evidence: list[dict]# 网络证据列表（完整字段）
    local_evidence: list[dict]  # 本地知识库证据列表

    # ── 证据裁判 (deep_dive_node) ──
    evidence_pool: list[dict]   # 汇聚后的证据池（含评分）
    deep_dive: str              # 裁判摘要
    audit: str                  # 审计摘要
    audit_flags: list[dict]     # 审计标记（低置信/冲突/缺失/过时）
    source_index: list[dict]    # 来源索引 [{source_id, label, locator, source_type}]

    # ── 分析 (analyze_node) ──
    analysis: str               # 分析摘要
    needs_more_research: bool   # 是否需要补搜
    missing_gaps: list[str]     # 缺失的信息维度
    supplementary_queries: list[dict]  # 补搜计划
    findings: list[dict]        # 研究结论 [{claim_id, dimension, claim, confidence, source_ids}]
    claim_map: list[dict]       # 结论与来源的映射

    # ── 检索统计（用于执行明细） ──
    web_retrieval_stats: dict   # {query_count, raw_count, kept_count, dropped_count}
    local_retrieval_stats: dict
    web_search_trace: list[dict]    # 每次检索的详细记录
    local_rag_trace: list[dict]

    # ── 输出 ──
    code: str               # 代码生成（已弃用）
    draft: str              # 报告草稿
    final: str              # 最终输出

    # ── 迭代控制 ──
    iteration: int          # 当前迭代轮次
    max_iterations: int     # 最大迭代轮次
```

## 2.2 Graph（工作流编排）

```python
# 文件: app/mult_agents/graph.py

def build_app(agents, checkpointer):
    """构建 LangGraph 工作流应用。
    
    Args:
        agents: AgentBundle（9 个已构建的 Agent 实例）
        checkpointer: 检查点存储器（用于断点续跑和记忆）
    
    Returns:
        编译后的 LangGraph 应用（可调用 .invoke(state, config)）
    """

    # ① 创建状态图
    workflow = StateGraph(ResearchState)

    # ② 注册节点（每个节点绑定一个 Agent 实例）
    workflow.add_node("intent",        bind_agent(intent_node,        agents.intent_router,    "intent_router"))
    workflow.add_node("direct_answer",  bind_agent(direct_answer_node, agents.direct_responder, "direct_responder"))
    workflow.add_node("plan",          bind_agent(plan_node,          agents.planner,          "planner"))
    workflow.add_node("web_search",    bind_agent(web_search_node,    agents.scout_web,        "scout_web"))
    workflow.add_node("local_rag",     bind_agent(local_rag_node,     agents.scout_local,      "scout_local"))
    workflow.add_node("deep_dive",     bind_agent(deep_dive_node,     agents.evidence_judge,   "evidence_judge"))
    workflow.add_node("analyze",       bind_agent(analyze_node,       agents.analyst,          "analyst"))
    workflow.add_node("reflect",       bind_agent(reflect_node,       agents.reflect_planner,  "reflect_planner"))
    workflow.add_node("write",         bind_agent(write_node,         agents.writer,           "writer"))

    # ③ 定义执行路径
    workflow.add_edge(START, "intent")

    # 意图分流
    workflow.add_conditional_edges("intent", route_after_intent, {
        "direct_answer": "direct_answer",  # 简单问答 → 直接回答 → END
        "plan": "plan",                     # 复杂研究 → 规划
    })

    # 规划 → 同时启动双源检索
    workflow.add_edge("plan", "web_search")
    workflow.add_edge("plan", "local_rag")

    # 双源检索 → 汇聚到证据裁判
    workflow.add_edge("web_search", "deep_dive")
    workflow.add_edge("local_rag", "deep_dive")

    # 证据裁判 → 分析
    workflow.add_edge("deep_dive", "analyze")

    # 分析 → 条件分流
    workflow.add_conditional_edges("analyze", should_continue_research, {
        "reflect": "reflect",  # 需要补搜 → 补搜规划 → 回到检索
        "write": "write"       # 证据足够 → 写报告
    })

    workflow.add_edge("reflect", "web_search")
    workflow.add_edge("reflect", "local_rag")
    workflow.add_edge("direct_answer", END)
    workflow.add_edge("write", END)

    return workflow.compile(checkpointer=checkpointer)
```

### 路由逻辑

```python
def route_after_intent(state) -> str:
    if state.get("intent") == "direct":
        return "direct_answer"
    return "plan"

def should_continue_research(state) -> str:
    iteration = state.get("iteration", 0)
    max_iter = state.get("max_iterations", 2)
    if iteration >= max_iter:       # 达到最大轮次 → 强制写报告
        return "write"
    if state.get("needs_more_research", False):  # 信息不足 → 补搜
        return "reflect"
    return "write"                  # 信息足够 → 写报告
```

### 9 个智能体角色一览

| Prompt Key | 角色名 | 职责 | 输出格式 | 使用的 Prompt 设计要点 |
|------------|--------|------|----------|----------------------|
| `intent_router` | IntentRouter | 意图分类 | JSON `{route,reason}` | 金融关键词触发 multiagent：分析/研究/估值/DCF/PE/PB/ROE... |
| `plan` | ChiefArchitect | 任务拆解 | JSON `{objective,sub_questions,outline,budget}` | 五层递进框架：宏观→行业→公司→估值→舆情 |
| `web_search` | WebScout | 网络取证 | JSON `{summary,evidence,gaps,rejected_source_ids}` | 6 级信源分层：official>institutional>media>data_provider>community>unknown |
| `local_rag` | LocalRAGScout | 知识库取证 | JSON 同 web_search | 文档权重：券商研报>白皮书>公告>新闻，3 个月内优先 |
| `deep_dive` | EvidenceJudge | 证据裁判 | JSON `{summary,evidence_pool,audit_flags,source_index}` | 5 级可信度评分（0.95→0.40）+ 交叉验证规则 + 利益冲突检测 |
| `analyze` | Analyst | 五维分析 | JSON `{analysis_summary,needs_more_research,findings,claim_map}` | 宏观/行业/公司/估值/舆情五维，每个结论绑定 source_ids |
| `reflect` | ResearchPlanner | 补搜规划 | JSON `{reflection_summary,supplementary_queries}` | 5 种补搜策略：换词/加限定/拆细/换源/交叉验证 |
| `write` | Writer | 研报撰写 | Markdown | 8 章结构（摘要→宏观→行业→公司→估值→舆情→风险→建议），3000+ 字 |
| `direct_answer` | DirectAnswer | 快速回答 | 纯文本 | 金融数据时效提醒 + 投资建议免责声明 |

## 2.6 完整执行时序

```
时间线（一次典型研究，约 25-35 秒）：

0s   │ intent_node
     │   └─ detect_intent("深度研究比亚迪002594")
     │      → 匹配关键词 "研究/分析" → multiagent
     │      → 跳过 LLM，0ms 返回  ← 【优化点】
     │
0s   │ plan_node
     │   └─ LLM 调用 (~3s)
     │      输入: "深度研究比亚迪002594"
     │      输出: {objective, sub_questions:[5个], outline:[7节], search_plan:[6条]}
     │
3s   │ web_search_node ┐
     │   ├─ bocha → DuckDuckGo (~2s) → 5 条网页结果
     │   ├─ collect_finance_data (~3s) → 8 条财务 + 8 条新闻
     │   ├─ raw_records = 21 条
     │   └─ LLM 过滤 (~3s) → web_evidence
     │                   ┤ 并行
     │ local_rag_node ──┘
     │   └─ Milvus 查询 (~1s) → 0 条（空库）
     │      → raw_records = 0 → 跳过 LLM  ← 【优化点】
     │
11s  │ deep_dive_node
     │   └─ LLM 调用 (~5s)  ← 数据最多的一步
     │      输入: web_evidence[] + local_evidence[]（压缩后）
     │      输出: evidence_pool[], source_index[], audit_flags[]
     │
16s  │ analyze_node
     │   └─ LLM 调用 (~4s)
     │      输入: evidence_pool[] + audit_flags[]
     │      输出: findings[], needs_more_research, missing_gaps
     │      判定: 信息充足 → 不需要补搜
     │
20s  │ write_node
     │   └─ LLM 调用 (~8s)  ← 生成 3000+ 字报告，最慢
     │      输入: findings[] + source_index[] + audit_flags[]
     │      输出: Markdown 研报
     │      → _validate_and_fix_citations() 校验引用
     │      → _ensure_reference_section() 拼接参考资料
     │
28s  │ 返回 final → SSE 流推前端
```
